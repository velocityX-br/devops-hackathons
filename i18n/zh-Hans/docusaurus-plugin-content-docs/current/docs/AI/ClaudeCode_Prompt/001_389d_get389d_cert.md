Here is my current script:
#!/usr/bin/env bash

# Function to select a garden if not provide in CLI or what provided is invalid
select_garden() {
  while true; do
    echo ""
    echo "Select the Gardener landscape:"
    echo "1) sap-landscape-live"
    echo "2) sap-landscape-canary"
    echo "3) sap-landscape-ac-live (china)"
    read -rp "Enter the number of your choice: " garden_choice

    case $garden_choice in
      1) GARDEN="sap-landscape-live"; break ;;
      2) GARDEN="sap-landscape-canary"; break ;;
      3) GARDEN="sap-landscape-ac-live"; break ;;
      *) echo "Invalid choice. Please select 1, 2, or 3..." ;;
    esac
  done
}

# Function to list shoots
get_shoots() {
  echo ""
  echo "Retrieving shoots in garden '$GARDEN', project 'sni'..."
  SHOOTS=($(kubectl get shoots -n garden-sni -o jsonpath='{range .items[*]}{.metadata.name}{"\n"}{end}'))
  if [ ${#SHOOTS[@]} -eq 0 ]; then
    echo "No shoots found in project 'sni'."
    exit 1
  fi
}

# Function to interactively select a shoot
select_shoot() {
  while true; do
    echo ""
    echo "Select a shoot cluster under garden '$GARDEN':"
    for i in "${!SHOOTS[@]}"; do
      printf "%2d) %s\n" $((i+1)) "${SHOOTS[$i]}"
    done
    read -rp "Enter the number of the shoot to target: " shoot_choice
    index=$((shoot_choice - 1))

    if [[ "$index" =~ ^[0-9]+$ ]] && [ "$index" -ge 0 ] && [ "$index" -lt "${#SHOOTS[@]}" ]; then
      SHOOT="${SHOOTS[$index]}"
      break
    else
      echo "Invalid selection. Please enter a valid number."
    fi
  done
}

# Main function
login() {
  if [[ -n "$1" ]]; then

    case ${INPUT_SEED#sap-landscape-} in

      canary) GARDEN="sap-landscape-canary" ;;
      livecn) GARDEN="sap-landscape-ac-live" ;;

      if [[ "$shoot" == "$INPUT_SHOOT" ]]; then

      fi

  echo ""
  echo "You can now use kubectl to interact with the shoot cluster '$SHOOT'."

## 生成结果

以下为 Claude Code 根据上述提示词生成的 `check_389ds_certs.sh` 脚本（原文保留）：

```
❯ cat check_389ds_certs.sh
#!/usr/bin/env bash

set -euo pipefail

GARDEN="sap-landscape-canary"
PROJECT="sni"
NAMESPACE="ldap"
POD="consumer-0"
PORT="60636"

# ─── Safety guard ─────────────────────────────────────────────────────────────
# Explicitly block live environments
block_live_envs() {
  local target_output
  target_output=$(gardenctl target 2>/dev/null || true)
  if echo "$target_output" | grep -qE "sap-landscape-live|sap-landscape-ac-live"; then
    echo "ERROR: Detected a live/ac-live garden in active target. Aborting for safety."
    exit 1
  fi
}

# ─── Prerequisite check ───────────────────────────────────────────────────────
if [[ -z "${KUBECONFIG:-}" ]]; then
  cat <<'EOF'
ERROR: missing kubectl-env, please add the following lines to your profile/bashrc:
  eval "$(gardenctl kubectl-env bash)"
EOF
  exit 1
fi

# ─── Confirmation prompt ──────────────────────────────────────────────────────
echo "======================================================"
echo "  389DS Certificate Expiry Check"
echo "======================================================"
echo ""
echo "This script will:"
echo "  1. Target garden : $GARDEN"
echo "  2. Iterate all shoots in project: $PROJECT"
echo "  3. For each shoot, check namespace '$NAMESPACE' exists"
echo "  4. If yes, exec into pod '$POD' and check cert expiry on port $PORT"
echo "  5. Output a summary report"
echo ""
echo "IMPORTANT: No delete or destructive commands will be run."
echo ""
read -rp "Proceed? (yes/no): " CONFIRM
if [[ "$CONFIRM" != "yes" ]]; then
  echo "Aborted."
  exit 0
fi

# ─── Safety check ─────────────────────────────────────────────────────────────
block_live_envs

# ─── Target canary garden ─────────────────────────────────────────────────────
echo ""
echo "Targeting garden '$GARDEN'..."
gardenctl target --garden "$GARDEN"

# ─── Get list of shoots ───────────────────────────────────────────────────────
echo "Retrieving shoots in project '$PROJECT'..."
IFS=$'\n' read -r -d '' -a SHOOTS < <(kubectl get shoots -n "garden-${PROJECT}" -o jsonpath='{range .items[*]}{.metadata.name}{"\n"}{end}' && printf '\0') || true

if [[ ${#SHOOTS[@]} -eq 0 ]]; then
  echo "No shoots found in project '$PROJECT' under garden '$GARDEN'."
  exit 1
fi

echo "Found ${#SHOOTS[@]} shoot(s): ${SHOOTS[*]}"
echo ""

# ─── Report arrays ────────────────────────────────────────────────────────────
declare -a COL_SHOOT=()
declare -a COL_LDAP_POD=()
declare -a COL_CERT_EXPIRY=()

# ─── Iterate over each shoot ──────────────────────────────────────────────────
for SHOOT in "${SHOOTS[@]}"; do
  echo "------------------------------------------------------"
  echo "Processing shoot: $SHOOT"

  # Target the shoot
  gardenctl target --garden "$GARDEN" --project "$PROJECT" --shoot "$SHOOT" 2>/dev/null

  # Re-export KUBECONFIG for the targeted shoot
  eval "$(gardenctl kubectl-env bash)" 2>/dev/null || true

  LDAP_STATUS="No"
  CERT_OUTPUT="N/A"

  # Check if ldap namespace exists
  if kubectl get namespace "$NAMESPACE" &>/dev/null; then
    echo "  [OK] Namespace '$NAMESPACE' found."
    LDAP_STATUS="Yes"

    # Check if pod exists and is running
    POD_PHASE=$(kubectl get pod "$POD" -n "$NAMESPACE" \
      -o jsonpath='{.status.phase}' 2>/dev/null || echo "NotFound")

    if [[ "$POD_PHASE" == "Running" ]]; then
      echo "  [OK] Pod '$POD' is Running. Checking certificate..."

      RAW=$(kubectl exec -i "$POD" -n "$NAMESPACE" -- sh -c \
        "echo | openssl s_client -connect localhost:${PORT} -showcerts 2>/dev/null \
         | openssl x509 -noout -text 2>/dev/null \
         | grep 'Not After'" 2>/dev/null || echo "ERROR: exec failed")

      # Trim whitespace
      CERT_OUTPUT=$(echo "$RAW" | sed 's/^[[:space:]]*//' | tr -s ' ')
      echo "  Cert expiry: $CERT_OUTPUT"
    else
      CERT_OUTPUT="Pod not Running (phase: $POD_PHASE)"
      echo "  [WARN] Pod '$POD' not Running: $POD_PHASE"
    fi
  else
    echo "  [SKIP] Namespace '$NAMESPACE' not found."
  fi

  COL_SHOOT+=("$SHOOT")
  COL_LDAP_POD+=("$LDAP_STATUS")
  COL_CERT_EXPIRY+=("$CERT_OUTPUT")
done

# ─── Print Markdown report ────────────────────────────────────────────────────
echo ""
echo "======================================================"
echo "  REPORT: 389DS Certificate Expiry — $GARDEN / $PROJECT"
echo "  Generated: $(date -u '+%Y-%m-%d %H:%M:%S UTC')"
echo "======================================================"
echo ""
printf "| %-40s | %-12s | %-60s |\n" "Shoot Cluster" "LDAP Pod?" "Certificate Not After"
printf "| %-40s | %-12s | %-60s |\n" "$(printf '%0.s-' {1..40})" "$(printf '%0.s-' {1..12})" "$(printf '%0.s-' {1..60})"

for i in "${!COL_SHOOT[@]}"; do
  printf "| %-40s | %-12s | %-60s |\n" \
    "${COL_SHOOT[$i]}" \
    "${COL_LDAP_POD[$i]}" \
    "${COL_CERT_EXPIRY[$i]}"
done

echo ""
echo "Done.
```

### 脚本行为摘要

1. **安全护栏**：检测当前 `gardenctl target` 是否指向 live / ac-live，若是则退出。
2. **交互确认**：执行前需输入 `yes` 才会继续。
3. **遍历 shoot**：在 canary 的 `sni` 项目下枚举所有 shoot 并逐个切换 context。
4. **证书检查**：若存在 `ldap` 命名空间且 `consumer-0` 为 Running，则 exec 执行 openssl 并解析 `Not After`。
5. **报告输出**：最终以 Markdown 表格打印 shoot 名称、是否存在 LDAP Pod、证书过期字段。
